#grfico de burbujas o grafico de bolas
def f(i,j):
    return (j**2)*(i**2)

x=[]
y=[]
z=[]
N=10
step=2
for i in np.arange(-N, N+step, step):
    for j in np.arange(-N, N+step, step):
        x.append(i)
        y.append(j)
        z.append(f(i,j))
        
plt.scatter(x, y, c=z, cmap='viridis',s=z, alpha=0.5)
plt.show()
